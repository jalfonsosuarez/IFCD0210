Funciones en Javascript

-> Estructura de una función:


        // Función tradicional
        function nombre_funcion( params ) {
            código de la función

            return valor_de_retorno
        }

        // Función de flecha
        const nombre_funcion = ( parámetros ) => {
            código de la función

            return valor_de_retorno
        }

Para ver el resultado de la ejecución del código, abre la consola del inspector.

-> Función para sumar dos números:


        function sumaDosNumeros( a, b ) {
            let total = a + b;
            return total;
        }
    
     sumaDosNumeros(5,6);

-> Función para saber cual de dos números es mayor:


        const mayorDeDosNumeros = ( primero, segundo ) => {
            let resultado = "";
            if ( primero === segundo ) {
                resultado = `${primero} y ${segundo} son iguales`;
            } else if( primero > segundo ) {
                resultado `${primero} es mayor que ${segundo}`;
            } else {
                resultado `$(segundo) es mayor que ${primero}`;
            }

            console.log(resultado);
            return resultado;
        }
    
     mayorDeDosNumeros(20,100);
     mayorDeDosNumeros(120,100);
     mayorDeDosNumeros(100,100);
