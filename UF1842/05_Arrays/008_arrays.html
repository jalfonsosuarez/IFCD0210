-> Array de letras a string:

    
        const arrayToString = (array) => {
            let cadena = '';
            for (let n=0; n < array.lenght; n++) {
                cadena += array[n];
            }           
            console.log(cadena);
        }
    
     arrayToString(['A','B','C',' ','D','E']);

-> Devolver el valor mínimo de un array numérico:

    
        const minimo = (listaNumeros) => {
            let minimo=Infinity;
            for (let n=0; n < listaNumeros.length; n++) {
                if (listaNumeros[n] < minimo) {
                    minimo = listaNumeros[n];
                }
            }           
            console.log(minimo);
        }
    
     minimo([10,30,15,6,8,100,3,-50]);

-> Devolver el producto de los positivos de un array:

    
        const producto = (listaNumeros) => {
            let salida = 1;
            for (let n=0; n < listaNumeros.length; n++) {
                if (listaNumeros[n] >= 0) {
                    salida *= listaNumeros[n];
                }
            }           
            console.log(salida);
        }
    
     producto([-2,4,-7,6,3]);
