Bucles for en Javascript

-> Estructura del bucle for:

        
            for (let index = 0; index < array.lenght; index++) {
                instrucciones del bucle;
                break;
                continue;
            }

-> Bucle for del uno al 10:

        
        for(let i=0; i<10; i++) {
            console.log(i);
        }
        
         delUnoAlDiez();

-> Bucle for doble: 

        
        for(let i=0, j=10; i<10; i++, j--) {
            console.log({i, j});
        }
        
         bucleDoble();

-> Uso de continue en bucle for:

        
        for(let i=0, j=10; i<10; i++, j--) {
            if(j === 5) {
                continue;
            }
            console.log({i, j});
        }
        
         forContinue();

-> Uso de breack en bucle for:

        
        for(let i=0, j=10; i<10; i++, j--) {
            if(i === 7) {
                break;
            }
            console.log({i, j});
        }            
        
         forBreak();

-> Bucles for anidados:

        
        for (let i=0; i<10; i++) {
            for (j=0; j<10; j++) {
                console.log({i,j});
            }
        }
        
         bucleAnidado();
