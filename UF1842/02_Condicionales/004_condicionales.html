Condicional en Javascript

if

edad = 19;
if (edad >= 18) {
  console.log('Es mayor de edad.');
}

if ... else

edad = 19;
if (edad >= 18) {
  console.log('Es mayor de edad.');
} else {
  console.log('Eres menor de edad.');
}

if ... else if .... else

nota = 9;
if (nota < 5) {
  console.log('SUSPENSO');
} else if ( nota < 6) {
  console.log('APROBADO');
} else if ( nota < 7 ) {
  console.log('BIEN');
} else if ( nota < 9 ) {
  console.log('NOTABLE');
} else {
  console.log('SOBRESALIENTE');
}

switch - case

nota = 6;
switch (nota) {
  case 10:
  case 9:
    console.log('SOBRESALIENTE');
    break;
  case 8:
  case 7:
    console.log('NOTABLE');
    break;
  case 6:
    console.log('BIEN');
    break;
  case 5:
    console.log('SUFICIENTE');
    break;
  default:
    console.log('SUSPENSO');
}
